3. Theme Service Toggle (theme.service.ts)
typescript
toggle(): void {
  this.isDark.update(current => {
    const newTheme = !current;                    // Flip boolean value
    localStorage.setItem('theme', newTheme ? 'dark' : 'light');  // Save to storage
    document.documentElement.classList.toggle('dark', newTheme); // Add/remove CSS class
    return newTheme;                              // Update signal value
  });
}

## Key Data Types:
